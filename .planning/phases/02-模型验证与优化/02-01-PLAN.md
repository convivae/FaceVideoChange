    ```python
    def get_model_path(model_name: str, models_dir: Path | None = None) -> Path | None:
        """
        Get path to a model file, checking multiple locations.
        
        Returns None if model not found.
        
        Args:
            model_name: Model name from MODEL_REGISTRY (e.g., 'inswapper_128_fp16')
            models_dir: Base models directory (default: DATA_DIR / 'models')
            
        Returns:
            Path to model file, or None if not found
        """
        if models_dir is None:
            from facevidechange.config import DATA_DIR
            models_dir = DATA_DIR / "models"
        
        if model_name not in MODEL_REGISTRY:
            return None
        
        info = MODEL_REGISTRY[model_name]
        
        # Check original directory
        path = models_dir / "original" / info.filename
        if path.exists():
            return path
        
        # Check optimized directory
        path = models_dir / "optimized" / info.filename
        if path.exists():
            return path
        
        # Check directly in models directory
        path = models_dir / info.filename
        if path.exists():
            return path
        
        return None


    def model_exists(model_name: str, models_dir: Path | None = None) -> bool:
        """Check if a model file exists in any expected location."""
        return get_model_path(model_name, models_dir) is not None
    ```

    Add to exports: get_model_path, model_exists
